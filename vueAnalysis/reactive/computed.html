computedеӨ„зҗҶ

еӨ„зҗҶcomputedзӣёе…ізҡ„йҖ»иҫ‘пјҢеҸ‘з”ҹеңЁinitStateдёӯпјҢжҺҘдёӢжқҘжҲ‘д»¬иҜҰз»ҶеҲҶжһҗдёҺcomputedзӣёе…ізҡ„йҖ»иҫ‘гҖӮ

export function initState (vm: Component) {
  // зңҒз•Ҙд»Јз Ғ
  const opts = vm.$options
  if (opts.computed) initComputed(vm, opts.computed)
}

жҲ‘д»¬зҹҘйҒ“computedи®Ўз®—еұһжҖ§жҳҜдҫқиө–дәҺе…¶е®ғе“Қеә”ејҸеҸҳйҮҸзҡ„пјҢеӣ жӯӨжҲ‘д»¬еҲҶжһҗcomputedзҡ„ж—¶еҖҷдјҡеҲҶдёәдёӨдёӘжӯҘйӘӨпјҡcomputedеҲқе§ӢеҢ–е’Ңcomputedжӣҙж–°гҖӮ

computedеҲқе§ӢеҢ–

еңЁinitState()ж–№жі•дёӯеҰӮжһңжҲ‘д»¬дј йҖ’дәҶcomputedпјҢйӮЈд№Ҳдјҡи°ғз”ЁinitComputed()ж–№жі•гҖӮinitComputed()ж–№жі•е®ҡд№үеңЁsrc/core/instance/state.jsж–Үд»¶дёӯпјҢе…¶д»Јз ҒеҰӮдёӢпјҡ

const computedWatcherOptions = { lazy: true }
function initComputed (vm: Component, computed: Object) {
  // $flow-disable-line
  const watchers = vm._computedWatchers = Object.create(null)
  // computed properties are just getters during SSR
  const isSSR = isServerRendering()
  for (const key in computed) {
    const userDef = computed[key]
    const getter = typeof userDef === 'function' ? userDef : userDef.get
    if (process.env.NODE_ENV !== 'production' && getter == null) {
      warn(
        `Getter is missing for computed property "${key}".`,
        vm
      )
    }
    if (!isSSR) {
      // create internal watcher for the computed property.
      watchers[key] = new Watcher(
        vm,
        getter || noop,
        noop,
        computedWatcherOptions
      )
    }
    // component-defined computed properties are already defined on the
    // component prototype. We only need to define computed properties defined
    // at instantiation here.
    if (!(key in vm)) {
      defineComputed(vm, key, userDef)
    } else if (process.env.NODE_ENV !== 'production') {
      if (key in vm.$data) {
        warn(`The computed property "${key}" is already defined in data.`, vm)
      } else if (vm.$options.props && key in vm.$options.props) {
        warn(`The computed property "${key}" is already defined as a prop.`, vm)
      }
    }
  }
}

  • еңЁinitComputed()ж–№жі•дёӯпјҢйҰ–е…Ҳе®ҡд№үдәҶдёҖдёӘ_computedWatchersзҡ„еҸҳйҮҸпјҢиҝҷдёӘеҸҳйҮҸзҡ„дҪңз”ЁжҳҜзј“еӯҳеҪ“еүҚе®һдҫӢжүҖжңүи®Ўз®—еұһжҖ§зҡ„watcherгҖӮ
  • жҺҘдёӢжқҘйҒҚеҺҶжүҖжңүзҡ„computedпјҢз„¶еҗҺеҜ№жҜҸдёҖдёӘcomputedиҝӣиЎҢзұ»еһӢеҲӨж–ӯпјҢеҰӮжһңжҳҜfunctionзұ»еһӢпјҢйӮЈд№ҲзӣҙжҺҘдҪҝз”ЁпјҢеҰӮжһңжҳҜеҜ№иұЎеҲҷд»ЈиЎЁжҳҜget/setеҪўејҸпјҢеҲҷзӣҙжҺҘеҸ–getгҖӮеҰӮжһңжңҖеҗҺиҺ·еҸ–еҲ°зҡ„computedзҡ„getterдёәnullпјҢеҲҷеңЁејҖеҸ‘зҺҜеўғдёӢжҸҗзӨәй”ҷиҜҜгҖӮ
// дёӨз§Қзұ»еһӢзҡ„и®Ўз®—еұһжҖ§
export default {
  props: ['index'],

  • жңҖеҗҺгҖҒйҰ–е…ҲеҲӨж–ӯдәҶеҪ“еүҚйҒҚеҺҶзҡ„computedжҳҜеҗҰе·Із»ҸеңЁvmе®һдҫӢдёҠпјҢеҰӮжһңдёҚеңЁеҲҷи°ғз”ЁdefineComputed()ж–№жі•пјҢеҰӮжһңеңЁиҝҳйңҖиҰҒеҲӨж–ӯеҪ“еүҚйҒҚеҺҶзҡ„computedжҳҜеҗҰе’ҢpropsгҖҒdataе‘ҪеҗҚеҶІзӘҒпјҢеҰӮжһңеҶІзӘҒеҲҷжҸҗзӨәй”ҷиҜҜгҖӮ

жіЁж„ҸпјҡеҜ№дәҺеӯҗз»„д»¶иҖҢиЁҖпјҢиҝҷдёӘж—¶еҖҷеҪ“еүҚйҒҚеҺҶзҡ„computedе·Із»ҸеңЁvmе®һдҫӢдёҠдәҶпјҢжүҖд»Ҙе№¶дёҚдјҡи°ғз”ЁdefineComputed()ж–№жі•пјҢжҲ‘д»¬д»ҺдёҠйқўд»Јз ҒжіЁйҮҠд№ҹиғҪзңӢзҡ„еҮәжқҘгҖӮеҜ№дәҺеӯҗз»„д»¶иҖҢиЁҖпјҢзңҹжӯЈinitComputedзҡ„иҝҮзЁӢжҳҜеҸ‘з”ҹеңЁVue.extendж–№жі•дёӯпјҡ

Vue.extend = function (extendOptions) {
  // зңҒз•Ҙд»Јз Ғ
  const Super = this
  const Sub = function VueComponent (options) {
    this._init(options)
  }
  Sub.prototype = Object.create(Super.prototype)
  Sub.prototype.constructor = Sub

  // еҲқе§ӢеҢ–еӯҗз»„д»¶зҡ„computed
  if (Sub.options.computed) {
    initComputed(Sub)
  }
}

// extend.jsдёӯзҡ„initComputedе®ҡд№ү
function initComputed (Comp) {
  const computed = Comp.options.computed
  for (const key in computed) {
    defineComputed(Comp.prototype, key, computed[key])
  }
}

з„¶еҗҺinitComputedи°ғз”Ёзҡ„defineComputed()ж–№жі•пјҢе°ұе’ҢжҲ‘д»¬зҺ°еңЁзҡ„defineComputed()ж–№жі•жҳҜеҗҢдёҖдёӘж–№жі•пјҢе®ғе’ҢжӯӨж—¶зҡ„initComputed()ж–№жі•е®ҡд№үеңЁеҗҢдёҖдёӘдҪҚзҪ®(src/core/instance/state.js)пјҡ

const sharedPropertyDefinition = {
  enumerable: true,
  configurable: true,
  get: noop,
  set: noop
}
export function defineComputed (
  target: any,
  key: string,
  userDef: Object | Function
) {
  const shouldCache = !isServerRendering()
  if (typeof userDef === 'function') {
    sharedPropertyDefinition.get = shouldCache
      ? createComputedGetter(key)
      : createGetterInvoker(userDef)
    sharedPropertyDefinition.set = noop
  } else {
    sharedPropertyDefinition.get = userDef.get
      ? shouldCache && userDef.cache !== false
        ? createComputedGetter(key)
        : createGetterInvoker(userDef.get)
      : noop
    sharedPropertyDefinition.set = userDef.set || noop
  }
  if (process.env.NODE_ENV !== 'production' &&
      sharedPropertyDefinition.set === noop) {
    sharedPropertyDefinition.set = function () {
      warn(
        `Computed property "${key}" was assigned to but it has no setter.`,
        this
      )
    }
  }
  Object.defineProperty(target, key, sharedPropertyDefinition)
}

defineComputed()ж–№жі•зҡ„йҖ»иҫ‘еҫҲз®ҖеҚ•пјҢж №жҚ®дёҚеҗҢзҡ„зұ»еһӢзҡ„computedпјҢеҜ№sharedPropertyDefinitionзҡ„getе’ҢsetиҝӣиЎҢиөӢеҖјгҖӮsharedPropertyDefinitionеңЁд№ӢеүҚзҡ„proxyдёӯпјҢжҲ‘д»¬е·Із»Ҹд»Ӣз»ҚиҝҮпјҢ е®ғе°ұжҳҜObject.defineProperty()ж–№жі•descriptorеҸӮж•°зҡ„дёҖдёӘе…ұдә«й…ҚзҪ®гҖӮ

еңЁйқһSSRжңҚеҠЎз«ҜжёІжҹ“зҡ„жғ…еҶөпјҢsharedPropertyDefinition.getзҡ„еҖјжҳҜи°ғз”ЁдәҶcreateComputedGetter()ж–№жі•пјҢиҖҢеңЁSSRжңҚеҠЎз«ҜжёІжҹ“зҡ„жғ…еҶөдёӢжҳҜи°ғз”ЁдәҶcreateGetterInvoker()ж–№жі•гҖӮжҲ‘д»¬еңЁеҲҶжһҗVueжәҗз Ғзҡ„иҝҮзЁӢдёӯпјҢеӣ дёәдҫ§йҮҚдәҺWebжөҸи§ҲеҷЁз«Ҝзҡ„иЎЁзҺ°пјҢеӣ жӯӨжҲ‘д»¬жҺҘдёӢжқҘдјҡеҲҶжһҗcreateComputedGetter()ж–№жі•зҡ„е®һзҺ°гҖӮcreateComputedGetter()ж–№жі•е’ҢdefineComputed()ж–№жі•е®ҡд№үеңЁеҗҢдёҖдёӘдҪҚзҪ®пјҢд»Јз ҒеҰӮдёӢпјҡ

function createComputedGetter (key) {
  return function computedGetter () {
    const watcher = this._computedWatchers && this._computedWatchers[key]
    if (watcher) {
      if (watcher.dirty) {
        watcher.evaluate()
      }
      if (Dep.target) {
        watcher.depend()
      }
      return watcher.value
    }
  }
}

жҲ‘д»¬еҸҜд»ҘзңӢеҲ°createComputedGetter()ж–№жі•иҝ”еӣһдәҶдёҖдёӘеҮҪж•°пјҢиҝҷдёӘеҮҪж•°дјҡеңЁиҺ·еҸ–computedзҡ„ж—¶еҖҷиў«и°ғз”ЁпјҢдҫӢеҰӮз»„д»¶жёІжҹ“зҡ„ж—¶еҖҷпјҡ

<template>
  <div>{{fullName}}</div>
</template>

ж №жҚ®д»ҘдёҠд»Јз ҒжҲ‘д»¬еҶҚжқҘзңӢcomputedGetter()ж–№жі•пјҡз»„д»¶жёІжҹ“зҡ„ж—¶еҖҷдјҡиҺ·еҸ–fullNameи®Ўз®—еұһжҖ§пјҢз„¶еҗҺи°ғз”ЁcomputedGetter()ж–№жі•пјҢеңЁиҝҷдёӘж–№жі•жү§иЎҢзҡ„ж—¶еҖҷпјҢйҰ–е…ҲеҲӨж–ӯwatcher.dirtyеұһжҖ§пјҢиҝҷдёӘеұһжҖ§еңЁnew Watcher()зҡ„ж—¶еҖҷдёҺжҲ‘д»¬дј е…Ҙзҡ„const computedWatcherOptions = { lazy: true }жңүе…ігҖӮеңЁWatcherзұ»зҡ„жһ„йҖ еҮҪж•°дёӯпјҢжңүиҝҷж ·дёҖж®өд»Јз Ғпјҡ

class Watcher {
  // зңҒз•Ҙд»Јз Ғ
  constructor (vm, expOrFn, cb, options, isRenderWatcher) {
    if (options) {
      this.lazy = !!options.lazy
    } else {
      this.lazy = false
    }
    this.dirty = this.lazy
  }
}

еӣ дёәжҲ‘д»¬дј е…Ҙзҡ„lazyеҖјдёәtrueпјҢеӣ жӯӨwatcher.dirtyжқЎд»¶еҲӨж–ӯдёәзңҹпјҢиҝӣиЎҢwatcher.evaluate()и®Ўз®—гҖӮйҡҸеҗҺеҲӨж–ӯдәҶDep.targetдёәзңҹпјҢеҲҷиҝӣиЎҢдҫқиө–ж”¶йӣҶwatcher.depend()пјҢе…ідәҺдҫқиө–ж”¶йӣҶжҲ‘д»¬дјҡеңЁд№ӢеҗҺзҡ„з« иҠӮиҜҰз»Ҷд»Ӣз»ҚгҖӮжҲ‘д»¬еҸӘиҰҒзҹҘйҒ“пјҢеҪ“еңЁз»„д»¶жёІжҹ“зҡ„ж—¶еҖҷи§ҰеҸ‘зҡ„computedдҫқиө–ж”¶йӣҶпјҢж”¶йӣҶзҡ„жҳҜrender watcherгҖӮжңҖеҗҺпјҢжҲ‘д»¬зңӢдёҖдёӢwatcher.evaluate()ж–№жі•зҡ„е®һзҺ°пјҡ

class Watcher {
  /**
   * Evaluate the value of the watcher.
   * This only gets called for lazy watchers.
   */
  evaluate () {
    this.value = this.get()
    this.dirty = false
  }
}

evaluate()ж–№жі•е®ғзҡ„е®һзҺ°йқһеёёз®ҖеҚ•пјҢе°ұжҳҜи§ҰеҸ‘computedзҡ„getterиҝӣиЎҢжұӮеҖјпјҢз„¶еҗҺжҠҠdirtyи®ҫзҪ®дёәfalseгҖӮ

computedжӣҙж–°

еңЁд»Ӣз»Қе®ҢдәҶcomputedзҡ„еҲқе§ӢеҢ–еҗҺпјҢжҲ‘д»¬еңЁжқҘзңӢcomputedзҡ„жӣҙж–°иҝҮзЁӢпјҢд»ҘдёӢйқўдҫӢеӯҗдёәдҫӢпјҡ

export default {
  template: `
    <div>{{fullName}}</div>
    <button @click="change">change</button>
  `
  data () {
    return {
      total: 0,
      firstName: 'first',

  computed: {
    fullName () {

    }

еӣ дёәtotalгҖҒfirstNameе’ҢlastNameе…ЁйғЁдёәе“Қеә”ејҸеҸҳйҮҸпјҢжүҖд»ҘfullNameиҝҷдёӘи®Ўз®—еұһжҖ§еҲқе§ӢеҢ–зҡ„ж—¶еҖҷпјҢжӯӨж—¶totalеҖјдёә0пјҢfullNameи®Ўз®—еұһжҖ§жңүдёӨдёӘWatcherпјҢе…¶дёӯдёҖдёӘжҳҜи®Ўз®—еұһжҖ§watcherпјҢеҸҰеӨ–дёҖдёӘжҳҜжёІжҹ“watcherгҖӮеҪ“зӮ№еҮ»жҢүй’®и§ҰеҸ‘дәӢд»¶еҗҺпјҢдјҡи§ҰеҸ‘totalеұһжҖ§зҡ„setterж–№жі•пјҢиҝӣиҖҢи°ғз”ЁдёҖдёӘеҸ«еҒҡnotifyзҡ„ж–№жі•гҖӮ

setпјҡ function reactiveSetter (newVal) {

е…¶дёӯnotify()жҳҜе®ҡд№үеңЁDepзұ»дёӯзҡ„дёҖдёӘж–№жі•пјҡ

export default class Dep {
   constructor () {
    this.subs = []
  }
  notify () {
    // stabilize the subscriber list first
    const subs = this.subs.slice()
    if (process.env.NODE_ENV !== 'production' && !config.async) {
      // subs aren't sorted in scheduler if not running async
      // we need to sort them now to make sure they fire in correct
      // order
      subs.sort((a, b) => a.id - b.id)
    }
    for (let i = 0, l = subs.length; i < l; i++) {
      subs[i].update()
    }
  }
}

д»Јз ҒеҲҶжһҗпјҡ

  • subsе°ұжҳҜжҲ‘д»¬ж”¶йӣҶиө·жқҘзҡ„watcherпјҢе®ғжҳҜдёҖдёӘж•°з»„пјҢеҜ№еә”дёҠйқўжЎҲдҫӢзҡ„иҜқе®ғжҳҜдёҖдёӘй•ҝеәҰдёә2зҡ„ж•°з»„е№¶дё”е…¶дёӯдёҖдёӘе…ғзҙ дёәrender watcherпјҢеҸҰдёҖдёӘе…ғзҙ дёәcomputed watcherгҖӮ
  • еңЁnotify()ж–№жі•и°ғз”Ёж—¶пјҢдјҡйҒҚеҺҶsubsж•°з»„пјҢз„¶еҗҺдҫқж¬Ўи°ғз”ЁеҪ“еүҚwatcherзҡ„updateж–№жі•гҖӮе…¶дёӯupdateж–№жі•жҳҜе®ҡд№үеңЁWatcherзұ»дёӯзҡ„дёҖдёӘе®һдҫӢж–№жі•пјҢд»Јз ҒеҰӮдёӢпјҡ
class Watcher {
  // зңҒз•Ҙе…¶е®ғ
  update () {
    /* istanbul ignore else */
    if (this.lazy) {
      this.dirty = true
    } else if (this.sync) {
      this.run()
    } else {
      queueWatcher(this)
    }
  }
}

еҪ“з¬¬дёҖж¬ЎйҒҚеҺҶж—¶пјҢжӯӨж—¶зҡ„watcherдёәи®Ўз®—еұһжҖ§watcherпјҢжҲ‘д»¬е·Із»ҸеңЁеүҚйқўд»Ӣз»ҚиҝҮи®Ўз®—еұһжҖ§watcherе®ғзҡ„this.lazyеҖјдёәtrueпјҢеӣ жӯӨдјҡиҝӣиЎҢthis.dirty = trueгҖӮ
